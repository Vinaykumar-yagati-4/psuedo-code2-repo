START
READ num
SET original = num
SET revNum = 0
WHILE num > 0
  SET digit = num % 10
  SET revNum = revNum * 10 + digit
  num = num / 10
ENDWHILE
IF original = revNum
PRINT "Palindrome"
ELSE
PRINT "Not palindrome"
END



ARMSTRONG NUMBER :

START 
READ num
SET sum = 0
WHILE num > 0
  SET digit = num % 10
  SET sum = sum + digit ^ digit
  num =num / 10
ENDWHILE
IF sum == digit
PRINT "Palindrome"
ELSE
PRINT "Not Palindrome"
END
